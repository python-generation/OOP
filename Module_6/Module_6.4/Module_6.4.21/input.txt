# TEST_6:
class ContextManager:
    def __init__(self):
        self.inside = False
  
    def __enter__(self):
        self.inside = True
        return self
  
    def __exit__(self, exc_type, exc_value, exc_tb):
        self.inside = False
        return True
        
        
print(is_context_manager(ContextManager()))
